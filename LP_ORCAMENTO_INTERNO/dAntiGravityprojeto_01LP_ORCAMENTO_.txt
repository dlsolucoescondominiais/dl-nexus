<script>
document.getElementById('orcForm').addEventListener('submit', async function(e) {
    e.preventDefault();
    const btn = document.getElementById('submitBtn');
    const msg = document.getElementById('statusMsg');
    btn.disabled = true;
    btn.innerText = '⏳ Processando...';
    msg.innerText = '';
    const fd = new FormData(this);
    // Coletar checkboxes agrupados
    const servicos = [...document.querySelectorAll('input[name="servicos"]:checked')].map(c => c.value);
    const chapa = [...document.querySelectorAll('input[name="chapa_detalhes"]:checked')].map(c => c.value);
    const modalidade = [...document.querySelectorAll('input[name="modalidade"]:checked')].map(c => c.value);
    fd.set('servicos_lista', JSON.stringify(servicos));
    fd.set('chapa_lista', JSON.stringify(chapa));
    fd.set('modalidade_lista', JSON.stringify(modalidade));
    const WEBHOOK_URL = 'https://n8n.dlsolucoescondominiais.com.br/webhook/orcamento-interno-dl';
    try {
        const res = await fetch(WEBHOOK_URL, { method: 'POST', body: fd });
        if (res.ok) {
            msg.style.color = '#00c853';
            msg.innerText = '✅ Orçamento enviado com sucesso! Aguarde o retorno via Telegram.';
            this.reset();
        } else {
            msg.style.color = '#ff5252';
            msg.innerText = '❌ Erro ao enviar. Verifique se o n8n está online.';
        }
    } catch (err) {
        msg.style.color = '#ff5252';
        msg.innerText = '❌ Falha na conexão com o servidor n8n.';
    } finally {
        btn.disabled = false;
        btn.innerText = '🚀 GERAR ORÇAMENTO E ENVIAR PARA N8N';
    }
});
</script>
